<?php
$backupDir = '/var/backups/disastershield';
$latestBackup = '';
$lastBackupTime = '';

// Get latest .tar.gz backup filename
if (is_dir($backupDir)) {
    $files = scandir($backupDir, SCANDIR_SORT_DESCENDING);
    foreach ($files as $file) {
        if (pathinfo($file, PATHINFO_EXTENSION) === 'gz' && strpos($file, '.tar.gz') !== false) {
            $latestBackup = $file;
            break;
        }
    }
}

// Get last readable timestamp from the text file
$timestampFile = '/var/www/html/last_backup.txt';
if (file_exists($timestampFile)) {
    $lastBackupTime = file_get_contents($timestampFile);
}
?>

<!DOCTYPE html>
<html>
<head>
    <title>DisasterShield - Backup Status</title>
    <style>
        body {
            font-family: Arial, sans-serif;
            background: #f9f9f9;
            padding: 40px;
            text-align: center;
        }
        .status-box {
            background: white;
            border-radius: 10px;
            display: inline-block;
            padding: 20px 40px;
            box-shadow: 0 0 10px rgba(0,0,0,0.1);
        }
        .status-box h2 {
            color: #333;
        }
    </style>
</head>
<body>
    <div class="status-box">
        <h2>üõ°Ô∏è Latest Backup Status</h2>
        <?php if ($latestBackup): ?>
            <p><strong>Backup File:</strong> <?= htmlspecialchars($latestBackup) ?></p>
            <p><strong>Location:</strong> /var/backups/disastershield/</p>
        <?php else: ?>
            <p>No backup archive found.</p>
        <?php endif; ?>

        <?php if ($lastBackupTime): ?>
            <p><strong><?= htmlspecialchars($lastBackupTime) ?></strong></p>
        <?php else: ?>
            <p>No backup timestamp recorded.</p>
        <?php endif; ?>
    </div>
</body>
</html>
